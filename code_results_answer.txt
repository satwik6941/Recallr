Here's a comprehensive Python code example to create an agentic Retrieval-Augmented Generation (RAG) system using LlamaIndex and OpenAI. This implementation includes document loading, indexing, querying, and agentic capabilities.

```python
import os
from typing import List, Dict, Any
from llama_index.core import (
    VectorStoreIndex,
    SimpleDirectoryReader,
    StorageContext,
    load_index_from_storage,
    Settings
)
from llama_index.core.node_parser import SentenceSplitter
from llama_index.core.tools import FunctionTool, QueryEngineTool
from llama_index.core.agent import ReActAgent
from llama_index.llms.openai import OpenAI
from llama_index.embeddings.openai import OpenAIEmbedding
from dotenv import load_dotenv

# Load environment variables
load_dotenv()

# Set up OpenAI API key
os.environ["OPENAI_API_KEY"] = os.getenv("OPENAI_API_KEY")

# Configure LlamaIndex settings
Settings.llm = OpenAI(model="gpt-4o-mini", temperature=0.1)
Settings.embed_model = OpenAIEmbedding(model="text-embedding-3-small")

class AgenticRAG:
    def __init__(self, documents_path: str = None, index_path: str = "./storage"):
        """
        Initialize the Agentic RAG system.

        Args:
            documents_path: Path to the directory containing documents to index
            index_path: Path to store/load the index
        """
        self.documents_path = documents_path
        self.index_path = index_path
        self.index = None
        self.agent = None

    def load_documents(self) -> List:
        """
        Load documents from the specified directory.

        Returns:
            List of loaded documents
        """
        if not self.documents_path:
            raise ValueError("Documents path not provided")

        print(f"Loading documents from {self.documents_path}")
        documents = SimpleDirectoryReader(self.documents_path).load_data()
        print(f"Loaded {len(documents)} documents")
        return documents

    def create_index(self, documents: List) -> VectorStoreIndex:
        """
        Create a vector index from the loaded documents.

        Args:
            documents: List of documents to index

        Returns:
            Created vector index
        """
        print("Creating index...")
        # Split documents into nodes
        node_parser = SentenceSplitter(chunk_size=1024, chunk_overlap=200)
        nodes = node_parser.get_nodes_from_documents(documents)

        # Create index
        self.index = VectorStoreIndex(nodes)
        print("Index created successfully")

        # Persist index to disk
        self.index.storage_context.persist(persist_dir=self.index_path)
        print(f"Index persisted to {self.index_path}")

        return self.index

    def load_index(self) -> VectorStoreIndex:
        """
        Load an existing index from storage.

        Returns:
            Loaded vector index
        """
        print(f"Loading index from {self.index_path}")
        storage_context = StorageContext.from_defaults(persist_dir=self.index_path)
        self.index = load_index_from_storage(storage_context)
        print("Index loaded successfully")
        return self.index

    def setup_agent(self) -> ReActAgent:
        """
        Set up the agent with tools for querying the index.

        Returns:
            Configured ReAct agent
        """
        if not self.index:
            raise ValueError("Index not created or loaded")

        print("Setting up agent...")

        # Create query engine
        query_engine = self.index.as_query_engine(similarity_top_k=3)

        # Create tools
        query_engine_tool = QueryEngineTool.from_defaults(
            query_engine=query_engine,
            name="document_query_engine",
            description="Useful for answering questions about the documents in the index"
        )

        # Create a simple calculator tool
        def multiply(a: float, b: float) -> float:
            """Multiply two numbers and returns the product"""
            return a * b

        multiply_tool = FunctionTool.from_defaults(
            fn=multiply,
            name="multiply_numbers",
            description="Useful for multiplying two numbers"
        )

        # Create agent
        self.agent = ReActAgent.from_tools(
            [query_engine_tool, multiply_tool],
            llm=Settings.llm,
            verbose=True
        )

        print("Agent setup complete")
        return self.agent

    def query(self, query_str: str) -> str:
        """
        Query the agent with a question.

        Args:
            query_str: The question to ask the agent

        Returns:
            The agent's response
        """
        if not self.agent:
            raise ValueError("Agent not set up")

        print(f"Querying agent with: {query_str}")
        response = self.agent.chat(query_str)
        return str(response)

def main():
    # Example usage
    documents_path = "./data"  # Path to your documents
    index_path = "./storage"  # Where to store the index

    # Initialize the Agentic RAG system
    rag_system = AgenticRAG(documents_path=documents_path, index_path=index_path)

    try:
        # Try to load existing index
        rag_system.load_index()
    except:
        # If no index exists, create a new one
        documents = rag_system.load_documents()
        rag_system.create_index(documents)

    # Set up the agent
    rag_system.setup_agent()

    # Example queries
    queries = [
        "What is the main topic of the documents?",
        "Summarize the key points in the documents",
        "What is 5 multiplied by 7?",
        "Combine the information from the documents with the multiplication result"
    ]

    for query in queries:
        print(f"\nQuery: {query}")
        response = rag_system.query(query)
        print(f"Response: {response}")

if __name__ == "__main__":
    main()
```
